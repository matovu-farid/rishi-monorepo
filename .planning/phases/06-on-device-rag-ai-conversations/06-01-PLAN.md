- `extractEpubText(filePath: string): Promise<Array<{ text: string; chapter: string | null }>>`:
  - Read EPUB file as base64 via `FileSystem.readAsStringAsync(filePath, { encoding: 'base64' })`
  - Use JSZip to unzip: `const zip = await JSZip.loadAsync(base64, { base64: true })`
  - Parse META-INF/container.xml to find rootfile path (look for `<rootfile` tag, extract `full-path` attribute)
  - Parse the OPF file to extract spine reading order: parse `<spine>` for `<itemref idref="...">`, then resolve each idref against `<manifest>` items to get href paths
  - For each spine item: load the XHTML file from the zip, strip all HTML tags using regex `/<[^>]*>/g`, collapse multiple whitespace to single space, trim
  - Extract chapter title: look for h1/h2/h3 content before stripping tags, or use the manifest item id as fallback
  - Return array of { text, chapter } sections in spine order
  - Handle errors gracefully: if a spine item is missing or parsing fails, skip it with console.warn

- `chunkText(sections: Array<{ text: string; chapter: string | null }>, maxChunkSize = 500, overlap = 50): TextChunk[]`:
  - Import `randomUUID` from `expo-crypto`
  - Split each section's text by sentence boundaries: `/(?<=[.!?])\s+/`
  - Accumulate sentences into a current chunk string until adding the next sentence would exceed maxChunkSize
  - When a chunk is full: push it as `{ id: randomUUID(), bookId: '', chunkIndex: n++, text: currentChunk.trim(), chapter, createdAt: Date.now() }`
  - Note: bookId is set to '' here; the caller (getChunks/pipeline) sets the real bookId
  - Implement overlap: after pushing a chunk, keep the last ~overlap characters as the start of the next chunk
  - After processing all sentences in a section, push any remaining text as a final chunk

- `getChunks(filePath: string, format: 'epub' | 'pdf'): Promise<TextChunk[]>`:
  - For EPUB: `const sections = await extractEpubText(filePath); return chunkText(sections)`
  - For PDF: `console.warn('[chunker] PDF text extraction not yet supported'); return []`

```typescript
import { rawDb } from '@/lib/db'

export function initVectorExtension(): void {
  // sqlite-vec bundled extension via withSQLiteVecExtension config plugin
  // The exact loading mechanism depends on expo-sqlite version:
  // Option A: rawDb.loadExtensionSync('vec0')
  // Option B: const ext = require('expo-sqlite').bundledExtensions['sqlite-vec']
  //           rawDb.loadExtensionSync(ext.libPath, ext.entryPoint)
  // Executor: check expo-sqlite docs for the correct pattern for SDK 54
  rawDb.loadExtensionSync('vec0')
}

export function ensureChunkTables(): void {
  rawDb.execSync(`
    CREATE TABLE IF NOT EXISTS chunks (
      id TEXT PRIMARY KEY,
      book_id TEXT NOT NULL,
      chunk_index INTEGER NOT NULL,
      text TEXT NOT NULL,
      chapter TEXT,
      created_at INTEGER NOT NULL
    )
  `)
  rawDb.execSync(`
    CREATE VIRTUAL TABLE IF NOT EXISTS chunk_vectors
    USING vec0(embedding float[384])
  `)
}

export function insertChunkWithVector(
  chunkId: string,
  bookId: string,
  chunkIndex: number,
  text: string,
  chapter: string | null,
  embedding: number[]
): void {
  const result = rawDb.runSync(
    'INSERT INTO chunks (id, book_id, chunk_index, text, chapter, created_at) VALUES (?, ?, ?, ?, ?, ?)',
    [chunkId, bookId, chunkIndex, text, chapter, Date.now()]
  )
  rawDb.runSync(
    'INSERT INTO chunk_vectors (rowid, embedding) VALUES (?, ?)',
    [result.lastInsertRowId, JSON.stringify(embedding)]
  )
}

export function searchSimilarChunks(
  bookId: string,
  queryEmbedding: number[],
  limit: number = 5
): Array<{ id: string; text: string; distance: number; chapter: string | null; chunkIndex: number }> {
  return rawDb.getAllSync(
    `SELECT c.id, c.text, c.chapter, c.chunk_index as chunkIndex, v.distance
     FROM chunk_vectors v
     INNER JOIN chunks c ON c.rowid = v.rowid
     WHERE v.embedding MATCH ?
       AND c.book_id = ?
     ORDER BY v.distance
     LIMIT ?`,
    [JSON.stringify(queryEmbedding), bookId, limit]
  ) as any
}

export function isBookEmbedded(bookId: string): boolean {
  const rows = rawDb.getAllSync(
    'SELECT 1 FROM chunks WHERE book_id = ? LIMIT 1',
    [bookId]
  )
  return rows.length > 0
}

export function deleteBookChunks(bookId: string): void {
  rawDb.execSync(
    `DELETE FROM chunk_vectors WHERE rowid IN (SELECT rowid FROM chunks WHERE book_id = '${bookId}')`
  )
  rawDb.execSync(`DELETE FROM chunks WHERE book_id = '${bookId}'`)
}
```
