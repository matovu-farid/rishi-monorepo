   ```typescript
   const handleEditNote = (highlight: Highlight) => {
     setPopoverVisible(false)
     setSelectedHighlight(highlight)
     noteEditorSheetRef.current?.snapToIndex(0)
   }
   const handleChangeColor = (highlightId: string, color: HighlightColor) => {
     updateHighlight(highlightId, { color })
     setHighlights(getHighlightsByBookId(book.id))
     setPopoverVisible(false)
     // Re-add annotation with new color: remove old, add new
     const h = highlights.find(h => h.id === highlightId)
     if (h) {
       removeAnnotation(h.cfiRange)
       const hex = HIGHLIGHT_COLORS.find(c => c.name === color)?.hex ?? '#FBBF24'
       addAnnotation('highlight', h.cfiRange, undefined, { id: h.id }, { color: hex, opacity: HIGHLIGHT_OPACITY })
     }
   }
   const handleDeleteHighlight = (highlightId: string) => {
     deleteHighlight(highlightId)
     const h = highlights.find(h => h.id === highlightId)
     if (h) removeAnnotation(h.cfiRange)
     setHighlights(getHighlightsByBookId(book.id))
     setPopoverVisible(false)
   }
   const handleSaveNote = (highlightId: string, note: string) => {
     updateHighlight(highlightId, { note: note || null })
     setHighlights(getHighlightsByBookId(book.id))
     noteEditorSheetRef.current?.close()
   }
   ```
